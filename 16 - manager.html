📍 Marker Information:

⚠️ Important: Marker Color Limitations

Google Maps CANNOT display yellow markers via URL.
Markers will appear as RED numbered pins when using Google Maps directly.
The numbers will show your Location ID if set, otherwise 1, 2, 3, etc.

✅ To Get Yellow Markers with White Borders:

  1. Use Google My Maps:
  2. Visit mymaps.google.com
  3. Create a new map and add your locations
  4. Click each marker → Style → Yellow color
  5. Add location numbers as labels
  6. Share the map and use that link for your QR code

This is the ONLY free way to get custom yellow markers!

📱 How This QR Code Works:

  1. User scans QR code
  2. Google Maps opens with numbered markers (red, not yellow)
  3. Each marker shows the location number/ID
  4. User taps a marker to see details
  5. Can get directions to that location

Backend Implementation Guide

🚀 Quick Setup with Node.js/Express

Initialize Backend Project

npm init -y npm install express cors body-parser mongodb dotenv npm install --save-dev nodemon

Create server.js

const express = require('express'); const cors = require('cors'); const bodyParser = require('body-parser'); const { MongoClient, ObjectId } = require('mongodb'); require('dotenv').config(); const app = express(); const PORT = process.env.PORT || 3000; // Middleware app.use(cors()); app.use(bodyParser.json()); // MongoDB Connection const uri = process.env.MONGODB_URI || 'mongodb://localhost:27017'; const client = new MongoClient(uri); let db; async function connectDB() { try { await client.connect(); db = client.db('location_manager'); console.log('Connected to MongoDB'); } catch (error) { console.error('MongoDB connection error:', error); } } // Routes // Get all locations app.get('/api/locations', async (req, res) => { try { const locations = await db.collection('locations').find({}).toArray(); res.json(locations); } catch (error) { res.status(500).json({ error: error.message }); } }); // Add new location app.post('/api/locations', async (req, res) => { try { const location = { ...req.body, counter: 0, createdAt: new Date(), updatedAt: new Date() }; const result = await db.collection('locations').insertOne(location); res.json({ ...location, _id: result.insertedId }); } catch (error) { res.status(500).json({ error: error.message }); } }); // Update location app.put('/api/locations/:id', async (req, res) => { try { const { id } = req.params; const update = { ...req.body, updatedAt: new Date() }; delete update._id; const result = await db.collection('locations').updateOne( { _id: new ObjectId(id) }, { $set: update } ); res.json({ success: result.modifiedCount > 0 }); } catch (error) { res.status(500).json({ error: error.message }); } }); // Delete location app.delete('/api/locations/:id', async (req, res) => { try { const { id } = req.params; const result = await db.collection('locations').deleteOne({ _id: new ObjectId(id) }); res.json({ success: result.deletedCount > 0 }); } catch (error) { res.status(500).json({ error: error.message }); } }); // Increment counter app.post('/api/locations/:id/increment', async (req, res) => { try { const { id } = req.params; const result = await db.collection('locations').updateOne( { _id: new ObjectId(id) }, { $inc: { counter: 1 }, $set: { lastVisit: new Date() } } ); res.json({ success: result.modifiedCount > 0 }); } catch (error) { res.status(500).json({ error: error.message }); } }); // Generate Maps URL app.get('/api/maps-url', async (req, res) => { try { const locations = await db.collection('locations').find({}).toArray(); // Generate Google Maps URL with multiple markers let mapsUrl = 'https://www.google.com/maps/dir/'; locations.forEach((loc, index) => { if (loc.coordinates) { mapsUrl += `${loc.coordinates.lat},${loc.coordinates.lng}/`; } else { mapsUrl += `${encodeURIComponent(loc.address)}/`; } }); res.json({ url: mapsUrl }); } catch (error) { res.status(500).json({ error: error.message }); } }); // Start server connectDB().then(() => { app.listen(PORT, () => { console.log(\`Server running on port \${PORT}\`); }); });

{ _id: ObjectId, name: String, address: String, number: String, // Location ID/Number phone: String, // Phone number link: String, // Venmo link counter: Number, // Visit counter coordinates: { lat: Number, lng: Number }, createdAt: Date, updatedAt: Date, lastVisit: Date }

Mobile Integration Endpoint

// Special endpoint for mobile QR code scanning app.get('/map', async (req, res) => { const locations = await db.collection('locations').find({}).toArray(); // Generate custom Google Maps URL const baseUrl = 'https://maps.google.com/maps?'; const params = new URLSearchParams(); // Add each location as a marker locations.forEach((loc, i) => { if (loc.coordinates) { params.append(\`markers\${i}\`, \`color:yellow|label:\${i+1}|\${loc.coordinates.lat},\${loc.coordinates.lng}\` ); } }); // Redirect to Google Maps res.redirect(baseUrl + params.toString()); });
